bca_planet_energy_job_add = {
	add = value:bca_base_energy
	add = modifier:planet_technician_energy_produces_add
	[[INC_ADD]
		add = $INC_ADD$
	]
}

bca_jobs_for_energy = {
	add = modifier:job_technician_add
	add = modifier:job_technician_drone_add
}

bca_unit_energy_production = {
	add = value:bca_base_energy
	add = modifier:planet_technician_energy_produces_add
	mult = value:bca_planet_energy_job_mult
}

bca_potential_standard_energy_production = {
	add = value:bca_jobs_for_energy
	[[INC_JOB] add = $INC_JOB$ ]
	divide = 100
	[[INC_ADD]
		mult = value:bca_planet_energy_job_add|INC_ADD|$INC_ADD$|
	]
	[[!INC_ADD]
		mult = value:bca_planet_energy_job_add
	]
	[[INC_MULT]
		mult = value:bca_planet_energy_job_mult|INC_MULT|$INC_MULT$|
	]
	[[!INC_MULT]
		mult = value:bca_planet_energy_job_mult
	]
}
