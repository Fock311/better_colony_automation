remove_zone_if_found = {
	if = {
		limit = {
			has_zone = { zone = $zone$ }
		}
		remove_zone = {
			zone = $zone$
		}
	}
}

replace_zone_in_district = {
	add_zone = {
		district = $district$
		zone = $zone$
		zone_slot = 1
		replace = yes
	}
}


remove_district_if = {
	if = {
		limit = {
			num_districts = { type = $district$ value > 0 }
			$cond$ = yes
		}
		remove_district = $district$
		remove_planet_flag = bca_pf_plan_remove_$district$
		remove_planet_flag = bca_pf_has_district_remove_plan
		remove_planet_flag = bca_pf_district_balanced
	}
}

debug_print_all_planet_flags = {
    log = "[Root.GetName] Planet ID: [Root.ID] - Starting flag dump"
    every_planet_flag = {     # 循环所有 flags（3.0+ 支持）
        log = "Flag: [This.GetValue]"  # This 是当前 flag
    }
    log = "Flag dump complete. Total flags: [Root.num_planet_flags]"
}
