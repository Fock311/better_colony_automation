remove_zone_if_found = {
	if = {
		limit = {
			has_zone = { zone = $zone$ }
		}
		remove_zone = {
			zone = $zone$
		}
	}
}

replace_zone_in_district = {
	add_zone = {
		district = $district$
		zone = $zone$
		zone_slot = 1
		replace = yes
	}
}


remove_district_if = {
	if = {
		limit = {
			num_districts = { type = $district$ value > 0 }
			$cond$ = yes
		}
		remove_district = $district$
		remove_planet_flag = bca_pf_plan_remove_$district$
		remove_planet_flag = bca_pf_has_district_remove_plan
		remove_planet_flag = bca_pf_district_balanced
	}
}
