try_set_secondary_districts_plan_zero_to_one = {
    if = {
        limit = {
             check_variable = {
                 which = bca_ps_plan_district_count_z3
                 value = 0
             }
            check_variable_arithmetic = {
                which = value:bca_num_free_secondary_districts_plan
                value > 0
            }
        }
        add_variable = {
            which = bca_ps_plan_district_count_z3
            value = 1
        }
    }
    if = {
        limit = {
             check_variable = {
