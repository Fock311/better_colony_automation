# test which district to build based on metric value comparison
# make shure that district_1 and district_2 can be added
# district_1 is preferred if metric_value is equal
bca_test_district_to_build = {
	add_district = { district_type = $district_1$ }
	set_variable = {
		which = bca_test_district_metric_1
		value = $metric_value$
	}
	remove_district = $district_1$
	add_district = { district_type = $district_2$ }
	set_variable = {
		which = bca_test_district_metric_2
		value = $metric_value$
	}
	remove_district = $district_2$
	if = {
		limit = {
			check_variable = {
				which = bca_test_district_metric_1
				value >= bca_test_district_metric_2
			}
		}
		bca_set_district_build_flag = {
			district = $district_1$
		}
	}
	else = {
		bca_set_district_build_flag = {
			district = $district_2$
		}
	}

}



bca_test_district_to_build_district_1_checked = {
	if = {
		limit = {
			has_district = $district_1$
			free_district_slots > 0
		}
		if = {
			limit = { 
				num_free_districts = { type = $district_2$ value = 0 }
				num_free_districts = { type = $district_1$ value > 0 } 
			}
			bca_set_district_build_flag = {
				district = $district_1$
			}
		}
		else_if = {
			limit = { 
				num_free_districts = { type = $district_1$ value = 0 }
				num_free_districts = { type = $district_2$ value > 0 }
			}	 
			bca_set_district_build_flag = {
				district = $district_2$
			}
		}
		else_if = {
			limit = {
				num_free_districts = { type = $district_1$ value > 0 }
				num_free_districts = { type = $district_2$ value > 0 }
			}
			bca_test_district_to_build = {
				district_1 = $district_1$
				district_2 = $district_2$
				metric_value = $metric_value$
			}
		}
	}
}

bca_test_district_to_replace = {
	if = {
		limit = {
			num_districts = { type = $district_replace_from$ value > $min_district_rf$ }
			NOT = { 
				bca_district_reach_planet_max = {
					type = $district_replace_to$
				}
			}
		}
		set_variable = {
			which = bca_test_district_metric_1
			value = $metric_value$
		}
		remove_district = $district_replace_from$
		add_district = { district_type = $district_replace_to$ }
		set_variable = {
			which = bca_test_district_metric_2
			value = $metric_value$
		}
		remove_district = $district_replace_to$
		add_district = { district_type = $district_replace_from$ }
		if = {
			limit = {
				check_variable = {
					which = bca_test_district_metric_2
					[[is_priority_district] value >= bca_test_district_metric_1 ]
					[[!is_priority_district] value > bca_test_district_metric_1 ]
				}
			}
			bca_set_district_remove_flag = {
				district = $district_replace_from$
			}
			bca_set_district_build_flag = {
				district = $district_replace_to$
			}
		}
		if = {limit = {always = $is_priority_district$}}
	}
}

bca_test_update_construction_metric_max_with_cost = {
	if = {
		limit = { 
			check_variable_arithmetic = {
				which = $production$
				subtract = bca_current_production
				divide = $cost$
				add = $priority_shift$
				value > bca_test_construction_metric_max
			}
		}
		set_variable = {
			which = bca_test_construction_metric_max
			value = $production$
		}
		subtract_variable = {
			which = bca_test_construction_metric_max
			value = bca_current_production
		}
		divide_variable = {
			which = bca_test_construction_metric_max
			value = $cost$
		}
		add_variable = {
			which = bca_test_construction_metric_max
			value = $priority_shift$
		}
		set_variable = {
			which = bca_test_construction_metric_max_index
			value = $index$
		}
	}
}

bac_get_num_free_building_slots_leq = {
	while = {
		count = @\[ $leq$ + 1 ]
		add_building = building_bca_test_$type$_slots
	}
	
	if = {
		limit = {
			num_buildings = { 
				type = building_bca_test_$type$_slots  
				value != @\[ $leq$ + 1 ]
			}
		}
		set_planet_flag = bca_free_$type$_slots_leq_$leq$
	}
	else = {
		remove_planet_flag = bca_free_$type$_slots_leq_$leq$
	}

	while = {
		count = @\[ $leq$ + 1 ]
		remove_building = building_bca_test_$type$_slots
	}
}

bac_get_num_free_foundry_building_slots_leq_2 = {
	while = {

	}

	while = {
		count = 3
		remove_building = building_bca_test_foundry_slots
	}
}

bac_test_zone_to_add = {
	export_trigger_value_to_variable = {
		trigger = num_zones
		parameters = {
			type = $zone$
		}
		variable = bca_test_counter_1
	}

	add_zone = {
		district = $district$
		zone = $zone$
		zone_slot = $zone_slot$
	}

	export_trigger_value_to_variable = {
		trigger = num_zones
		parameters = {
			type = $zone$
		}
		variable = bca_test_counter_2
	}

	if = {
		limit = {
			check_variable = { which = bca_test_counter_1 value = bca_test_counter_2 }
		}
		set_planet_flag = bca_pf_failed_to_add_$zone$
	}
	else = {
		bca_test_update_construction_metric_max_with_cost = {
			production = $production$
			cost = $cost$
			priority_shift = $priority_shift$
			index = $index$
		}

		remove_zone = {
			district = $district$
			zone = $zone$
			zone_slot = $zone_slot$
		}
	}
}

			}
		}
		remove_planet_flag = bca_pf_failed_to_add_$zone$
	}
}
