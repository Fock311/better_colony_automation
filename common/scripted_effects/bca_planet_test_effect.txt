bca_test_update_construction_metric_max_with_cost = {
	if = {
		limit = { 
			check_variable_arithmetic = {
				which = $production$
				subtract = bca_current_production
				divide = $cost$
				add = $priority_shift$
				value > bca_test_construction_metric_max
			}
		}
		set_variable = {
			which = bca_test_construction_metric_max
			value = $production$
		}
		subtract_variable = {
			which = bca_test_construction_metric_max
			value = bca_current_production
		}
		divide_variable = {
			which = bca_test_construction_metric_max
			value = $cost$
		}
		add_variable = {
			which = bca_test_construction_metric_max
			value = $priority_shift$
		}
		set_variable = {
			which = bca_test_construction_metric_max_index
			value = $index$
		}
	}
}

bca_get_num_free_city_zones = {
	add_zone = {
		district = 0
		zone = zone_bca_test
		zone_slot = 1
	}
	if = {
		limit = {
			has_upgraded_capital = yes
		}
		add_zone = {
			district = 0
			zone = zone_bca_test
			zone_slot = 2
		}
	}

	export_trigger_value_to_variable = {
		trigger = num_zones
		parameters = {
			type = zone_bca_test
		}
		variable = bca_num_free_city_zones
	}

	remove_zone = {
		zone = zone_bca_test
	}
	remove_zone = {
		zone = zone_bca_test
	}
}
